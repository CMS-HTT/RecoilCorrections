MET recoil corrections

----- How to set up the MET Recoil Correction Interface ------

> cd ${CMSSW_BASE}/src
> cmsenv
> git clone https://github.com/CMS-HTT/RecoilCorrections.git HTT-utilities/RecoilCorrections 
> scram b 

---- How to use the code ------------

// add the header file to your source file
#include "HTT-utilities/RecoilCorrections/interface/RecoilCorrector.h"

...

// create instances of class RecoilCorrection and
// load recoil resolution functions before looping over events
// the path to files is defined relative to #CMSSW_BASE/src directory

RecoilCorrector recoilPFMetCorrector("HTT-utilities/RecoilCorrections/data/PFMET_MG_2016BCD_RooT_5.2.root"); // Type-I PF MET 
RecoilCorrector recoilMvaMetCorrector("HTT-utilities/RecoilCorrections/data/MvaMET_MG_2016BCD_RooT_5.2.root"); // MVA MET
 
...

// apply recoil corrections for each event

// recoil corrections by quantile mapping --->
recoilPFMetCorrector.Correct(
    pfmet_ex,     // uncorrected met px
    pfmet_ey,     // uncorrected met py
    genZPx,       // generator Z(W) px
    genZPy,       // generator Z(W) py
    visZPx,       // visible (generator) Z(W) px
    visZPy,       // visible (generator) Z(W) py
    nJets30,      // number of jets (pT > 30 GeV, |eta|<4.7, Loose PFJet Id)
    pfmetcorr_ex, // corrected met px
    pfmetcorr_ey  // corrected met py
);
recoilMvaMetCorrector.Correct(
    mvamet_ex, 
    mvamet_ey, 
    genZPx,    
    genZPy,    
    visZPx,    
    visZPy,    
    nJets30,   
    mvametcorr_ex,
    mvametcorr_ey 
);

// recoil corrections by rescaling -->
recoilPFMetCorrector.CorrectByMeanResolution(
    pfmet_ex,     // uncorrected met px
    pfmet_ey,     // uncorrected met py
    genZPx,       // generator Z(W) px
    genZPy,       // generator Z(W) py
    visZPx,       // visible (generator) Z(W) px
    visZPy,       // visible (generator) Z(W) py
    nJets30,      // number of jets (pT > 30 GeV, |eta|<4.7, Loose PFJet Id)
    pfmetcorr_ex, // corrected met px
    pfmetcorr_ey  // corrected met py
);
recoilMvaMetCorrector.CorrectByMeanResolution(
    mvamet_ex, 
    mvamet_ey, 
    genZPx,    
    genZPy,    
    visZPx,    
    visZPy,    
    nJets30,   
    mvametcorr_ex,
    mvametcorr_ey 
);
